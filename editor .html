export default { async fetch(request, env) { const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type", }; // ১. CORS OPTIONS হ্যান্ডেল করা if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } const url = new URL(request.url); const PROJECT_ID = "tournament-97743"; const BASE_FS_URL = `https://firestore.googleapis.com/v1/projects/${PROJECT_ID}/databases/(default)/documents`; // ============================================================ // ২. GET SECTION: অ্যাপ এখান থেকে মেটাডাটা রিড করবে // ============================================================ if (request.method === "GET") { let key = ""; if (url.pathname.includes("/sliders")) key = "SLIDER_DATA"; else if (url.pathname.includes("/videos")) key = "VIDEO_DATA"; else if (url.pathname.includes("/withdraw_methods")) key = "METHOD_DATA"; else if (url.pathname.includes("/state")) key = "STATE_DATA"; else if (url.pathname.includes("/app_settings")) key = "SETTING_DATA"; if (!key) { return new Response(JSON.stringify({ error: "Invalid path" }), { status: 404, headers: corsHeaders }); } const data = await env.KV3.get(key); return new Response(data || "[]", { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // ============================================================ // ৩. POST SECTION: অ্যাডমিন প্যানেল থেকে ডাটা সেভ বা সিঙ্ক // ============================================================ if (request.method === "POST") { try { const body = await request.json(); const type = body.sync_type; let kvKey = ""; if (type === "sliders") kvKey = "SLIDER_DATA"; else if (type === "videos") kvKey = "VIDEO_DATA"; else if (type === "withdraw_methods") kvKey = "METHOD_DATA"; else if (type === "app_settings") kvKey = "SETTING_DATA"; else if (type === "state") kvKey = "STATE_DATA"; if (!kvKey) throw new Error("Invalid sync_type"); // --- A. ডাইরেক্ট ডাটা আপলোড (Admin uploads JSON directly) --- if (body.direct_data) { await env.KV3.put(kvKey, JSON.stringify(body.direct_data)); return new Response(JSON.stringify({ success: true, message: type + " saved directly to KV" }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // --- B. ফায়ারস্টোর থেকে সিঙ্ক (বর্তমানে শুধু State/Leaderboard এর জন্য) --- let firestoreUrl = ""; if (type === "state") { firestoreUrl = `${BASE_FS_URL}/profile?pageSize=10&orderBy=balance%20desc`; } else { throw new Error("Sync from Firestore is only supported for 'state' type"); } const fsRes = await fetch(firestoreUrl); if (!fsRes.ok) throw new Error("Firestore connect failed: " + fsRes.statusText); const fsData = await fsRes.json(); const finalData = (fsData.documents || []).map(doc => { const parsed = parseFirestoreFields(doc.fields); if (type === "state") { return { name: parsed.fullName || "Gamer", balance: parsed.balance || 0, img: parsed.photoURL || "" }; } return parsed; }); await env.KV3.put(kvKey, JSON.stringify(finalData)); return new Response(JSON.stringify({ success: true, message: type + " Sync Successful", count: finalData.length }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (e) { return new Response(JSON.stringify({ success: false, error: e.message }), { status: 500, headers: corsHeaders }); } } } }; // ============================================================ // ৪. HELPER FUNCTION: ফায়ারস্টোর ডাটা পার্সার // ============================================================ function parseFirestoreFields(fields) { let res = {}; if (!fields) return res; for (let key in fields) { const val = fields[key]; if (val.stringValue !== undefined) res[key] = val.stringValue; else if (val.integerValue !== undefined) res[key] = parseInt(val.integerValue); else if (val.doubleValue !== undefined) res[key] = parseFloat(val.doubleValue); else if (val.booleanValue !== undefined) res[key] = val.booleanValue; else if (val.arrayValue) { res[key] = (val.arrayValue.values || []).map(v => v.mapValue ? parseFirestoreFields(v.mapValue.fields) : (v.stringValue || v.integerValue) ); } else if (val.mapValue) res[key] = parseFirestoreFields(val.mapValue.fields); } return res; }
