export default { async fetch(request, env) { // ১. CORS কনফিগারেশন (যাতে যেকোনো অ্যাপ থেকে রিকোয়েস্ট আসতে পারে) const corsHeaders = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Methods": "GET, POST, OPTIONS", "Access-Control-Allow-Headers": "Content-Type", }; // ২. ব্রাউজারের প্রি-ফ্লাইট চেক (OPTIONS) হ্যান্ডেল করা if (request.method === "OPTIONS") { return new Response(null, { headers: corsHeaders }); } const url = new URL(request.url); // ৩. ইউজারের অ্যাপ থেকে ডাটা চাইলে (GET Request) if (request.method === "GET") { let key = "TOURNAMENT_DATA"; // রাউটিং লজিক: ভবিষ্যতে লিডারবোর্ড বা অন্য কিছু চাইলে এখান থেকে ডাইভার্ট হবে if (url.pathname.includes("/state")) key = "STATE_DATA"; if (url.pathname.includes("/config")) key = "APP_CONFIG"; const data = await env.KV2.get(key); return new Response(data || "[]", { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } // ৪. অ্যাডমিন প্যানেল থেকে সিঙ্ক সিগন্যাল আসলে (POST Request) if (request.method === "POST") { const FIRESTORE_URL = "https://firestore.googleapis.com/v1/projects/tournament-97743/databases/(default)/documents/game_data/tournament_hub"; try { const response = await fetch(FIRESTORE_URL); if (!response.ok) throw new Error("Firestore connection failed: " + response.statusText); const data = await response.json(); // ফায়ারস্টোর ডাটা পার্সিং if (!data.fields || !data.fields.active_matches) { throw new Error("Firestore document is empty or formatted incorrectly"); } const rawMatches = data.fields.active_matches.arrayValue.values || []; const cleanMatches = rawMatches.map(m => parseFirestoreFields(m.mapValue.fields)); // KV2-তে ডাটা সেভ করা await env.KV2.put("TOURNAMENT_DATA", JSON.stringify(cleanMatches)); return new Response(JSON.stringify({ success: true, message: "Sync Successful", count: cleanMatches.length }), { headers: { ...corsHeaders, "Content-Type": "application/json" } }); } catch (e) { return new Response(JSON.stringify({ success: false, error: e.message }), { status: 500, headers: corsHeaders }); } } } }; // ফায়ারস্টোর ডাটা পার্সার (Helper Function) function parseFirestoreFields(fields) { let res = {}; for (let key in fields) { const val = fields[key]; if (val.stringValue) res[key] = val.stringValue; else if (val.integerValue) res[key] = parseInt(val.integerValue); else if (val.doubleValue) res[key] = parseFloat(val.doubleValue); else if (val.booleanValue !== undefined) res[key] = val.booleanValue; else if (val.arrayValue) { res[key] = (val.arrayValue.values || []).map(v => v.mapValue ? parseFirestoreFields(v.mapValue.fields) : (v.stringValue || v.integerValue) ); } else if (val.mapValue) res[key] = parseFirestoreFields(val.mapValue.fields); } return res; }
